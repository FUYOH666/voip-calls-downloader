# Конфигурация для Stranzit Audio Downloader
# ⚠️  ВАЖНО: Скопируйте этот файл в .env и заполните своими данными
# ⚠️  НИКОГДА не коммитьте .env файл в git!

# === АУТЕНТИФИКАЦИЯ ===
# Ваши логин и пароль для входа в личный кабинет lk.stranzit.ru
STRANZIT_USERNAME=ваш_логин_здесь
STRANZIT_PASSWORD=ваш_пароль_здесь

# === НАСТРОЙКИ ЗАГРУЗКИ ===
# Директория для сохранения скачанных аудио файлов (относительно корня проекта)
DOWNLOAD_DIR=./downloads

# Интервал проверки новых звонков в секундах (300 = 5 минут)
CHECK_INTERVAL=300

# === ФИЛЬТРЫ ЗАПИСЕЙ ===
# Период для поиска звонков:
#   - Формат: ДД.ММ.ГГГГ ЧЧ:ММ (например: 01.01.2025 00:00)
#   - Или ключевые слова: today_start (начало сегодня) / now (сейчас)
CALL_FILTER_START=today_start
CALL_FILTER_END=now

# Направление звонков: any (любые) / incoming (входящие) / outgoing (исходящие)
CALL_FILTER_DIRECTION=incoming

# Фильтр по длительности (опционально):
#   Оператор: >= (больше или равно), <= (меньше или равно), == (равно)
CALL_FILTER_DURATION_OP=>=
# Значение длительности в формате ЧЧ:ММ:СС (например: 00:03:00 для 3 минут)
CALL_FILTER_DURATION=00:03:00

# Количество записей на страницу при запросе (рекомендуется 50)
CALL_FILTER_RECORDS_PER_PAGE=50

# === ДОПОЛНИТЕЛЬНЫЕ НАСТРОЙКИ ===
# Режим отладки (true/false)
DEBUG=false
# Уровень логирования (DEBUG/INFO/WARNING/ERROR)
LOG_LEVEL=INFO

# === УВЕДОМЛЕНИЯ ОБ ОШИБКАХ (опционально) ===
# Webhook для отправки уведомлений об ошибках
# WEBHOOK_URL=https://your-webhook-url.com/endpoint

# Email уведомления об ошибках
# SMTP_SERVER=smtp.gmail.com
# SMTP_USER=your-email@gmail.com
# SMTP_PASS=your-app-password
# ALERT_EMAIL=alerts@your-domain.com

# === БАЗА ДАННЫХ ===
# Путь к файлу базы данных (автоматически создается при первом запуске)
DATABASE_PATH=./stranzit_calls.db
